function integrate_trapzoidal(f::Function, a::Number, b::Number, n::Integer)
    a, b = minmax(a, b)
    h = (b-a)/n
    x = range(a, b, length = n+1)
    ff = f.(x)
    result = 0.5*(ff[1]+ff[end])
    result += sum(ff[2:end-1])
    return result*h
end

function integrate_simpson_1_3(f::Function, a::Number, b::Number, n::Integer)
    @assert n %2 ==0
    a, b = minmax(a, b)
    h = (b-a)/n
    x = range(a, b, length = n+1)
    ff = f.(x)
    result = (ff[1]+ff[end])
    result += 4*sum(ff[2:2:end-1])
    result += 2*sum(ff[3:2:end-2])
    return result * h/3
end

function integrate_simpson_3_8(f::Function, a::Number, b::Number, n::Integer)
    @assert n %3 ==0
    a, b = minmax(a, b)
    h = (b-a)/n
    x = range(a, b, length = n+1)

    result = (ff[1]+ff[end])
    result += 3*sum(ff[2:3:end-2])
    result += 3*sum(ff[3:3:end-1])
    result += 2*sum(ff[4:3:end-3])
    return result * h * 3 / 8
end
